import matplotlib.pyplot as plt

labels = ["東部", "南部", "北部", "中部"]
sizes = [5, 10, 20, 15]
colors = ["red", "green", "blue", "yellow"]
explode = (0, 0, 0.05, 0)
plt.pie(sizes,explode = explode,labels = labels,colors = colors,\
    labeldistance = 1.1,autopct = "%2.1f%%",shadow = True,\
    startangle = 90,pctdistance = 0.6)
plt.axis("equal")
plt.legend()
plt.show()

from bokeh.io import show, output_file
from bokeh.models import ColumnDataSource
from bokeh.palettes import Spectral6
from bokeh.plotting import figure
from bokeh.transform import factor_cmap

output_file("bar_colormapped.html")

fruits = ['Apples', 'Pears', 'Nectarines', 'Plums', 'Grapes', 'Strawberries']
counts = [1,5,7,9,13,16]

source = ColumnDataSource(data=dict(fruits=fruits, counts=counts))

p1 = figure(x_range=fruits, plot_height=350, toolbar_location=None, title="Fruit Counts")
p1.vbar(x='fruits', top='counts', width=0.9, source=source, legend="fruits",
       line_color='white', fill_color=factor_cmap('fruits', palette=Spectral6, factors=fruits))

p1.xgrid.grid_line_color = None
p1.y_range.start = 0
p1.y_range.end = 20
p1.legend.orientation = "horizontal"
p1.legend.location = "top_center"

show(p1)
